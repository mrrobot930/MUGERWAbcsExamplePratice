public class L01_P06_Hypot {
    public static void main(String[] args) {
        double adjacent = 5.5;
        double opposite = 3.25;
        double hypotenuse = Math.sqrt(Math.pow(adjacent, 2) + Math.pow(opposite, 2));
        System.out.println("The hypotenuse of the right-angle triangle is: " + hypotenuse);
    }

}
